2020-02-28 【代码仓库】 

#include<iostream>
#include<cmath>
#include<cstring>
using namespace std;
int c[100005]; 
int f(char ch1,char ch2){
	int x=ch1-96,y=ch2-96;
	return min(abs(x-y),min(abs(x+26-y),abs(x-y-26)));
} 
int main(){
	int n,k,ans=0;
	string s;
	cin>>n>>k>>s;
	k--;
	for(int i=0;i<n;i++){
		c[i]=f(s[i],s[n-i-1]);
	}
	int L=-1,R=-1;
	if(k<n/2){
		for(int i=0;i+i<=n;i++)if(c[i]){L=i;break;}
		for(int i=n/2-1;i>=0;i--)if(c[i]){R=i;break;}
		if(L==R&&L==-1){
			cout<<0;
			return 0;
		}
		if(k<=L){
			ans+=R-k;
		}
		else if(k>=R){
			ans+=k-L;
		}
		else{
			ans+=R-L+min(k-L,R-k); 
		}
		for(int i=L;i<=R;i++){
			ans+=c[i];
		}
		cout<<ans;
	}
	else{
		for(int i=n/2;i<=n;i++)if(c[i]){L=i;break;}
		for(int i=n;i>=n/2;i--)if(c[i]){R=i;break;}
		if(L==R&&L==-1){
			cout<<0;
			return 0;
		}
		if(k<=L){
			ans+=R-k;
		}
		else if(k>=R){
			ans+=k-L;
		}
		else{
			ans+=R-L+min(k-L,R-k); 
		}
		for(int i=L;i<=R;i++){
			ans+=c[i];
		}
		cout<<ans;
		
	}
	return 0;
}
