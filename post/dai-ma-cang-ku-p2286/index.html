#include<iostream>
#include<cstdlib>
using namespace std;
#define MAXN 1000005
#define int long long
struct fhq_treap{
	int rt,cnt,siz[MAXN],ch[MAXN][2],rnd[MAXN],val[MAXN];
	int new_node(int v){
		cnt++;
		rnd[cnt]=rand();
		val[cnt]=v;
		siz[cnt]=1;
		return cnt;
	}
	void up(int root){
		siz[root]=siz[ch[root][0]]+siz[ch[root][1]]+1;
	}
	int merge(int x,int y){
		if(!x||!y)return x+y;
		if(rnd[x]<rnd[y]){
			ch[x][1]=merge(ch[x][1],y);
			up(x);
			return x;
		}
		else{
			ch[y][0]=merge(x,ch[y][0]);
			up(y);
			return y;
		}
	}
	void split(int root,int pos,int &x,int &y){//<pos:x?y
		if(!root){
			x=y=0;
			return;
		}
		if(val[root]<pos){
			x=root;
			split(ch[root][1],pos,ch[root][1],y);
		}
		else{
			y=root;
			split(ch[root][0],pos,x,ch[root][0]);
		}
		up(root);
	}
	void insert(int val){
		int x,y=new_node(val),z;
		split(rt,val,x,z);
		rt=merge(merge(x,y),z);
	}
	void del(int val){
		int x,y,z;
		split(rt,val,x,y);
		split(y,val+1,y,z);
		y=merge(ch[y][0],ch[y][1]);
		rt=merge(merge(x,y),z);
	}
	int get_kth(int root,int k){
		while(1){
			if(siz[ch[root][0]]>=k)root=ch[root][0];
			else if(siz[ch[root][0]]==k-1)return val[root];
			else{
				k-=siz[ch[root][0]]+1;
				root=ch[root][1];
			}
		}
	}
	int get_rank(int v){
		int x,y,ans;
		split(rt,v,x,y);
		ans=siz[x]+1;
		rt=merge(x,y);
		return ans;
	}
	int pre(int v){
		int x,y,ans;
		split(rt,v,x,y);
		if(siz[x]==0)return -1000000000;
		ans=get_kth(x,siz[x]);
		rt=merge(x,y);
		return ans;
	}
	int suf(int v){
		int x,y,ans;
		split(rt,v+1,x,y);
		if(siz[y]==0)return 1000000000;
		ans=get_kth(y,1);
		rt=merge(x,y);
		return ans;
	}
}T; 
signed main(){
	int n,opt,x,t1,t2,ans=0;
	int kk=2;
	cin>>n;
	for(int i=0;i<n;i++){
		cin>>opt>>x;
		if(opt==0){
			if(kk==2)kk=0;
			if(kk==0){
				T.insert(x);
			}
			else{
				t1=T.pre(x);
				t2=T.suf(x);
				if(t2-x<x-t1){
					T.del(t2);
					ans+=t2-x;
				}
				else{
					T.del(t1);
					ans+=x-t1;
				}
			}
			if(T.siz[T.rt]==0)kk=2;
		}
		else if(opt==1){
			if(kk==2)kk=1;
			if(kk==1){
				T.insert(x);
			}
			else{
				t1=T.pre(x);
				t2=T.suf(x);
				if(t2-x<x-t1){
					T.del(t2);
					ans+=t2-x;
				}
				else{
					T.del(t1);
					ans+=x-t1;
				}
			}
			if(T.siz[T.rt]==0)kk=2;
		}
		ans%=1000000;
	}
	cout<<ans<<endl;
	return 0;
}
