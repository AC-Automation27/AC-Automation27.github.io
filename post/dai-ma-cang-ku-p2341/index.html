#include<iostream>
#include<stack>
using namespace std;
struct egde{
	int v,nxt;
}a[100005];
int head[50005],cnt;
void add_edge(int u,int v){
	cnt++;
	a[cnt].nxt=head[u];
	a[cnt].v=v;
	head[u]=cnt;
}
int dfn[50005],low[50005],clr[50005],cnt2,tim;
bool in[50005];
stack<int> stk;
void tarjan(int u){
	in[u]=true;
	dfn[u]=++tim;
	low[u]=tim;
	stk.push(u);
	for(int i=head[u];i;i=a[i].nxt){
		if(!dfn[a[i].v]){
			tarjan(a[i].v);
			low[u]=min(low[u],low[a[i].v]);
		}
		else if(in[a[i].v])low[u]=min(low[u],dfn[a[i].v]);
	}
	if(dfn[u]==low[u]){
		cnt2++;
		int k;
		while(1){
			clr[k=stk.top()]=cnt2;
			stk.pop();
			if(k==u)break;
		}
	}
}
int du[50005];
int main(){
	int n,m,u,v;
	cin>>n>>m;
	for(int i=0;i<m;i++){
		cin>>u>>v;
		add_edge(u,v);
	}
	for(int i=1;i<=n;i++)
		if(!dfn[i])tarjan(i);
	for(int i=1;i<=n;i++){
		for(int j=head[i];j;j=a[j].nxt){
			if(clr[a[j].v]!=clr[i])du[clr[i]]++;
		}
	}
	//for(int i=1;i<=cnt2;i++)cout<<du[i]<<' ';
	int sum=0,k;
	for(int i=1;i<=cnt2;i++){
		if(du[i]==0)sum++,k=i;
	}
	if(sum>1||sum==0)cout<<0;
	else{
		sum=0;
		for(int i=1;i<=n;i++)if(clr[i]==k)sum++;
		cout<<sum;
	}
	return 0;
}
