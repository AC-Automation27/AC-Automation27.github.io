#include<iostream>
using namespace std;
int a[5000005],b[5000005],sum[5000005];
int main(){
	int n,x,y;
	cin>>n;
	for(int i=2;i<=5000000;i++){
		if(a[i]==0){
			for(int j=i*2;j<=5000000;j+=i){
				a[j]=i;
			}
		}
	}
	for(int i=2;i<=5000000;i++){
		if(a[i]==0)b[i]=1;
		else{
			b[i]=b[a[i]]+b[i/a[i]];
		}
	}
	for(int i=2;i<=5000000;i++){
	//	cout<<b[i]<<' ';
		sum[i]=sum[i-1]+b[i];
	}
	for(int i=0;i<n;i++){
		cin>>x>>y;
		cout<<sum[x]-sum[y]<<endl;
	} 
	return 0;
}
