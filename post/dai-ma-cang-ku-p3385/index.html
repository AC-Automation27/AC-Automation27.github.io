#include<iostream>
using namespace std;
int head[4005],cnt,dis[4005],q[10000005],hd,tl,c[4005];
bool vis[4005]; 
struct node{
	int v,nxt,val;
}a[10005];
void add_edge(int u,int v,int val){
	cnt++;
	a[cnt].v=v;
	a[cnt].val=val;
	a[cnt].nxt=head[u];
	head[u]=cnt; 
}
int main(){
	int n,m,u,v,w;
	int T;
	cin>>T;
	while(T--){
        hd=tl=0;
		cin>>n>>m;
		cnt=0;
		for(int i=1;i<=n;i++){
			dis[i]=1000000000;
			head[i]=0;
			vis[i]=0;
			c[i]=0;
		}
		for(int i=0;i<m;i++){
			cin>>u>>v>>w;
			add_edge(u,v,w);
			if(w>=0)add_edge(v,u,w);
		}
		q[tl++]=1;
		dis[1]=0;
		bool bj=0;
		while(hd<tl){
			int u=q[hd];
			for(int i=head[u];i;i=a[i].nxt){
				if(dis[a[i].v]>dis[u]+a[i].val){
					dis[a[i].v]=dis[u]+a[i].val;
					c[a[i].v]++;
					if(!vis[a[i].v]){
						q[tl++]=a[i].v;
						vis[a[i].v]=1;
					}
					if(c[a[i].v]==n){
						bj=1;
						break;
					}
				}
			}
			if(bj==1)break;
			vis[u]=0;
			hd++;
		}
		if(bj==1)cout<<"YE5\n";
		else cout<<"N0\n";
	}
	return 0;
}
