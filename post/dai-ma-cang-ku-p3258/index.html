#include<iostream>
using namespace std;
int head[500005],cnt=1;
int depth[500005],son[500005],siz[500005],top[500005],fa[500005];
struct node{
	int v,nxt;
}a[1000005];
void add_edge(int u,int v){
	a[cnt].v=v;
	a[cnt].nxt=head[u];
	head[u]=cnt; 
	cnt++;
}
void dfs1(int u,int f){
	depth[u]=depth[f]+1;
	fa[u]=f;
	siz[u]=1;
	int maxx=0,tmp=0;
	for(int i=head[u];i;i=a[i].nxt){
		if(a[i].v==f){
			continue; 
		}
		dfs1(a[i].v,u);
		siz[u]+=siz[a[i].v];
		if(siz[a[i].v]>maxx){
			maxx=siz[a[i].v];
			tmp=a[i].v;
		}
	}
	son[u]=tmp;
}
void dfs2(int u,int htop){
	top[u]=htop;
	if(son[u])
		dfs2(son[u],htop);
	for(int i=head[u];i;i=a[i].nxt){
		if(a[i].v==fa[u]||a[i].v==son[u])continue;
		dfs2(a[i].v,a[i].v);
	}
}
int lca(int x,int y){
	while(top[x]!=top[y]){
		if(depth[top[x]]<depth[top[y]])swap(x,y);
		x=fa[top[x]];
	}
	if(depth[x]<depth[y])swap(x,y);
	return y;
}
int q[500005],p[500005],r[500005];
void dfs(int u){
	r[u]=p[u];
	for(int i=head[u];i;i=a[i].nxt){
		if(a[i].v==fa[u])continue;
		dfs(a[i].v);
		r[u]+=r[a[i].v];
	}
	return;
}
int main(){
	ios::sync_with_stdio(false); 
	int n,u,v,temp;
	cin>>n;
	for(int i=1;i<=n;i++)cin>>q[i];
	for(int i=1;i<n;i++){
		cin>>u>>v;
		add_edge(u,v);
		add_edge(v,u);
	}
	dfs1(1,0);
	dfs2(1,1);
	for(int i=1;i<n;i++){
		p[q[i]]++;
		p[q[i+1]]++;
		int LCA=lca(q[i],q[i+1]);
		p[LCA]--;
		p[fa[LCA]]--;
	}
	dfs(1);
	for(int i=2;i<=n;i++)r[q[i]]--;
	for(int i=1;i<=n;i++)cout<<r[i]<<endl;
	return 0;
}
