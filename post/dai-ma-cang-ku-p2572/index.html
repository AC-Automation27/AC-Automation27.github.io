#include<iostream>
using namespace std;
bool data[100005];
struct node{
	int sum,tag,lmax[2],rmax[2],omax[2];
	bool rev;
} a[400005];
node up(node lft,node rgt,int lln,int rln){
	node ret;
	ret.sum=lft.sum+rgt.sum;
	if(lft.lmax[0]==lln)ret.lmax[0]=lft.lmax[0]+rgt.lmax[0];
	else ret.lmax[0]=lft.lmax[0];
	if(rgt.rmax[0]==rln)ret.rmax[0]=rgt.rmax[0]+lft.rmax[0];
	else ret.rmax[0]=rgt.rmax[0];
	ret.omax[0]=max(max(lft.omax[0],rgt.omax[0]),lft.rmax[0]+rgt.lmax[0]);
	
	if(lft.lmax[1]==lln)ret.lmax[1]=lft.lmax[1]+rgt.lmax[1];
	else ret.lmax[1]=lft.lmax[1];
	if(rgt.rmax[1]==rln)ret.rmax[1]=rgt.rmax[1]+lft.rmax[1];
	else ret.rmax[1]=rgt.rmax[1];
	ret.omax[1]=max(max(lft.omax[1],rgt.omax[1]),lft.rmax[1]+rgt.lmax[1]);
	ret.rev=0;
	ret.tag=2;
	return ret;
}
int ls(int x){return x<<1;}
int rs(int x){return (x<<1)+1;}
void pd(int rt,int l,int r){
	int mid=(l+r)>>1;
	if(a[rt].tag!=2){
		a[ls(rt)].tag=a[rt].tag;
		a[rs(rt)].tag=a[rt].tag;
		a[ls(rt)].rev=a[rs(rt)].rev=0;
		a[rt].tag=2;
		a[ls(rt)].omax[1]=a[ls(rt)].rmax[1]=a[ls(rt)].lmax[1]=a[ls(rt)].tag*(mid-l+1);
		a[ls(rt)].omax[0]=a[ls(rt)].rmax[0]=a[ls(rt)].lmax[0]=(1-a[ls(rt)].tag)*(mid-l+1);
		a[ls(rt)].sum=a[ls(rt)].tag*(mid-l+1);
		
		a[rs(rt)].omax[1]=a[rs(rt)].rmax[1]=a[rs(rt)].lmax[1]=a[rs(rt)].tag*(r-mid);
		a[rs(rt)].omax[0]=a[rs(rt)].rmax[0]=a[rs(rt)].lmax[0]=(1-a[rs(rt)].tag)*(r-mid);
		a[rs(rt)].sum=a[rs(rt)].tag*(r-mid);
		a[rt].rev=0;
	}
	if(a[rt].rev==1){
			a[ls(rt)].rev^=1;
			if(a[ls(rt)].tag!=2)a[ls(rt)].tag^=1;
			swap(a[ls(rt)].lmax[0],a[ls(rt)].lmax[1]);
			swap(a[ls(rt)].rmax[0],a[ls(rt)].rmax[1]);
			swap(a[ls(rt)].omax[0],a[ls(rt)].omax[1]);
			a[ls(rt)].sum=mid-l+1-a[ls(rt)].sum;
			a[rs(rt)].rev^=1;
			if(a[rs(rt)].tag!=2)a[rs(rt)].tag^=1;
			swap(a[rs(rt)].lmax[0],a[rs(rt)].lmax[1]);
			swap(a[rs(rt)].rmax[0],a[rs(rt)].rmax[1]);
			swap(a[rs(rt)].omax[0],a[rs(rt)].omax[1]);
			a[rs(rt)].sum=r-mid-a[rs(rt)].sum;
		a[rt].rev=0;
	}
}
void build(int rt,int l,int r){
	a[rt].rev=0;
	a[rt].tag=2;
	if(l==r){
		a[rt].sum=data[l];
		a[rt].omax[1]=a[rt].rmax[1]=a[rt].lmax[1]=(data[l]==1);
		a[rt].omax[0]=a[rt].rmax[0]=a[rt].lmax[0]=(data[l]==0);
		return;
	}
	int mid=(l+r)>>1;
	build(ls(rt),l,mid);
	build(rs(rt),mid+1,r);
	a[rt]=up(a[ls(rt)],a[rs(rt)],mid-l+1,r-mid);
	return;
}
void alc(int rt,int l,int r,int ql,int qr,bool k){
	if(ql<=l&&qr>=r){
		a[rt].tag=k;
		a[rt].rev=0;
		a[rt].omax[1]=a[rt].rmax[1]=a[rt].lmax[1]=k*(r-l+1);
		a[rt].omax[0]=a[rt].rmax[0]=a[rt].lmax[0]=(1-k)*(r-l+1);
		a[rt].sum=k*(r-l+1);
		return;		
	}
	int mid=(l+r)>>1;
	pd(rt,l,r);
	if(ql<=mid)alc(ls(rt),l,mid,ql,qr,k);
	if(qr>mid)alc(rs(rt),mid+1,r,ql,qr,k);
	a[rt]=up(a[ls(rt)],a[rs(rt)],mid-l+1,r-mid);
	return;
}
void swp(int rt,int l,int r,int ql,int qr){
	if(ql<=l&&qr>=r){
		if(a[rt].tag!=2)a[rt].tag^=1;
		a[rt].rev^=1;
		swap(a[rt].lmax[0],a[rt].lmax[1]);
		swap(a[rt].rmax[0],a[rt].rmax[1]);
		swap(a[rt].omax[0],a[rt].omax[1]);
		a[rt].sum=r-l+1-a[rt].sum;
		return;		
	}
	int mid=(l+r)>>1;
	pd(rt,l,r);
	if(ql<=mid)swp(ls(rt),l,mid,ql,qr);
	if(qr>mid)swp(rs(rt),mid+1,r,ql,qr);
	a[rt]=up(a[ls(rt)],a[rs(rt)],mid-l+1,r-mid);
	return;
}
int query_sum(int rt,int l,int r,int ql,int qr){
	if(ql<=l&&qr>=r){
		return a[rt].sum;
	}
	int mid=(l+r)>>1,s=0;
	pd(rt,l,r);
	if(ql<=mid)s+=query_sum(ls(rt),l,mid,ql,qr);
	if(qr>mid)s+=query_sum(rs(rt),mid+1,r,ql,qr);
	a[rt]=up(a[ls(rt)],a[rs(rt)],mid-l+1,r-mid);
	return s;
}
node query(int rt,int l,int r,int ql,int qr){
	pd(rt,l,r);
	if(ql<=l&&qr>=r){
		return a[rt];
	}
	int mid=(l+r)>>1;
	node s;
	bool x=0;
	if(ql<=mid){
		s=query(ls(rt),l,mid,ql,qr);
		x=1;
	}
	if(qr>mid){
		if(x==1)s=up(s,query(rs(rt),mid+1,r,ql,qr),mid-l+1,r-mid);
		else s=query(rs(rt),mid+1,r,ql,qr);
	}
	a[rt]=up(a[ls(rt)],a[rs(rt)],mid-l+1,r-mid);
	return s;
}
int main(){
	int n,m,opt,l,r;
	cin>>n>>m;
	for(int i=1;i<=n;i++)cin>>data[i];
	build(1,1,n);
	for(int i=0;i<m;i++){
		cin>>opt>>l>>r;
		l++;r++;
		if(opt==0)alc(1,1,n,l,r,0);
		else if(opt==1)alc(1,1,n,l,r,1);
		else if(opt==2)swp(1,1,n,l,r);
		else if(opt==3)cout<<query_sum(1,1,n,l,r)<<endl;
		else if(opt==4)cout<<query(1,1,n,l,r).omax[1]<<endl;
	}
	return 0;
}
