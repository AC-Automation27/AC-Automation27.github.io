#include<iostream>
using namespace std;
int head[50005],dis[50005],queue[10000005],p[50005],hd,tl,cnt;
struct node{
	int v,nxt,val;
}a[1000005];
void add_edge(int u,int v,int val){
	a[++cnt].v=v;
	a[cnt].val=val;
	a[cnt].nxt=head[u];
	head[u]=cnt;
}
int main(){
	int n,m,u,v,w,opt;
	cin>>n>>m;
	for(int i=0;i<m;i++){
		cin>>opt>>u>>v;
		if(opt==3){
			add_edge(u,v,0);
			add_edge(v,u,0);
			continue;
		}
		cin>>w;
		if(opt==1)
			add_edge(u,v,-w);
		else
			add_edge(v,u,w);
	}
	for(int i=1;i<=n;i++){
		add_edge(n+1,i,0);
		dis[i]=100000000;
	}
	queue[tl++]=n+1;
	while(hd<tl){
		u=queue[hd];
		hd++;
		for(int i=head[u];i;i=a[i].nxt){
			if(dis[u]+a[i].val<dis[a[i].v]){
				dis[a[i].v]=dis[u]+a[i].val;
				queue[tl++]=a[i].v;
				p[a[i].v]++;
				if(p[a[i].v]==n){
					cout<<"No";
					return 0;
				} 
			}
		}
	}
	cout<<"Yes"; 
	return 0;
}
