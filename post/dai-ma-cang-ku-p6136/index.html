#include<iostream>
#include<cstdlib>
#include<cstdio>
using namespace std;
#define MAXN 1100005
struct fhq_treap{
	int rt,cnt,siz[MAXN],ch[MAXN][2],rnd[MAXN],val[MAXN];
	inline int new_node(int v){
		cnt++;
		rnd[cnt]=rand();
		val[cnt]=v;
		siz[cnt]=1;
		return cnt;
	}
	inline void up(int root){
		siz[root]=siz[ch[root][0]]+siz[ch[root][1]]+1;
	}
	int merge(int x,int y){
		if(!x||!y)return x+y;
		if(rnd[x]<rnd[y]){
			ch[x][1]=merge(ch[x][1],y);
			up(x);
			return x;
		}
		else{
			ch[y][0]=merge(x,ch[y][0]);
			up(y);
			return y;
		}
	}
	void split(int root,int pos,int &x,int &y){//<pos:x?y
		if(!root){
			x=y=0;
			return;
		}
		if(val[root]<pos){
			x=root;
			split(ch[root][1],pos,ch[root][1],y);
		}
		else{
			y=root;
			split(ch[root][0],pos,x,ch[root][0]);
		}
		up(root);
	}
	inline void insert(int val){
		int x,y=new_node(val),z;
		split(rt,val,x,z);
		rt=merge(merge(x,y),z);
	}
	inline void del(int val){
		int x,y,z;
		split(rt,val,x,y);
		split(y,val+1,y,z);
		y=merge(ch[y][0],ch[y][1]);
		rt=merge(merge(x,y),z);
	}
	inline int get_kth(int root,int k){
		while(1){
			if(siz[ch[root][0]]>=k)root=ch[root][0];
			else if(siz[ch[root][0]]==k-1)return val[root];
			else{
				k-=siz[ch[root][0]]+1;
				root=ch[root][1];
			}
		}
	}
	inline int get_rank(int v){
		int x,y,ans;
		split(rt,v,x,y);
		ans=siz[x]+1;
		rt=merge(x,y);
		return ans;
	}
	inline int pre(int v){
		int x,y,ans;
		split(rt,v,x,y);
		ans=get_kth(x,siz[x]);
		rt=merge(x,y);
		return ans;
	}
	inline int suf(int v){
		int x,y,ans;
		split(rt,v+1,x,y);
		ans=get_kth(y,1);
		rt=merge(x,y);
		return ans;
	}
}T; 
inline int read(){
	int x=0;
	char ch=getchar();
	while(ch<'0'||ch>'9')ch=getchar();
	while(ch>='0'&&ch<='9'){
		x*=10;
		x+=ch-48;
		ch=getchar();
	}
	return x;
}
int main(){
	int n,m,opt,x,lst=0,ans=0;
	cin>>n>>m;
	for(register int i=0;i<n;i++){
		opt=read();
		T.insert(opt);
	}
	for(register int i=0;i<m;i++){
		opt=read();
		x=read();
		x^=lst;
		if(opt==1){
			T.insert(x);
		}
		else if(opt==2){
			T.del(x);
		}
		else if(opt==3){
			lst=T.get_rank(x);
			ans^=lst;
		}
		else if(opt==4){
			lst=T.get_kth(T.rt,x);
			ans^=lst;
		}
		else if(opt==5){
			lst=T.pre(x);
			ans^=lst;
		}
		else if(opt==6){
			lst=T.suf(x);
			ans^=lst;
		}
	}
	cout<<ans<<endl;
	return 0;
}
