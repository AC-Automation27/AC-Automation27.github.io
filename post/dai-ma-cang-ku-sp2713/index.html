#include<iostream>
#include<cmath> 
using namespace std;
#define ll long long
ll data[100005];
struct node{
	bool gt;
	ll sum;
} a[400005];
inline int ls(int x){return x<<1;}
inline int rs(int x){return (x<<1)+1;}
void up(int rt){
	a[rt].sum=a[ls(rt)].sum+a[rs(rt)].sum;
	a[rt].gt=a[ls(rt)].gt&a[rs(rt)].gt;
	return;
}
void build(int rt,int l,int r){
	if(l==r){
		a[rt].gt=(data[l]<=1);
		a[rt].sum=data[l];
		return;
	}
	int mid=(l+r)>>1;
	build(ls(rt),l,mid);
	build(rs(rt),mid+1,r);
	up(rt);
	return;
}
void change(int rt,int l,int r,int ql,int qr){
	if(ql<=l&&qr>=r&&a[rt].gt==1)return;
	if(l==r){
		a[rt].sum=sqrt(a[rt].sum);
		if(a[rt].sum<=1)a[rt].gt=1;
		return;
	}
	int mid=(l+r)>>1;
	if(ql<=mid)change(ls(rt),l,mid,ql,qr);
	if(qr>mid)change(rs(rt),mid+1,r,ql,qr);
	up(rt);
	return; 
}
ll query(int rt,int l,int r,int ql,int qr){
	if(ql<=l&&qr>=r)return a[rt].sum;
	int mid=(l+r)>>1;
	ll ans=0;
	if(ql<=mid)ans+=query(ls(rt),l,mid,ql,qr);
	if(qr>mid)ans+=query(rs(rt),mid+1,r,ql,qr);
	return ans;
}
int main(){
	ios::sync_with_stdio(false);
	cin.tie(0);
	cout.tie(0);
	int n,m,x,y,z,ttt=0;
	while(cin>>n){
		ttt++; 
		cout<<"Case #"<<ttt<<":"<<endl; 
		for(int i=1;i<=n;i++)cin>>data[i];
		build(1,1,n);
		cin>>m;
		for(register int i=0;i<m;i++){
			cin>>x>>y>>z;
			if(y>z)swap(y,z);
			if(x==0)change(1,1,n,y,z);
			else cout<<query(1,1,n,y,z)<<endl;
		}
		cout<<endl;
	}
	return 0;
}
