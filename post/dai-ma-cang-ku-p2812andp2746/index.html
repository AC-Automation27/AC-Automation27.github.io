#include<iostream>
#include<stack>
using namespace std;
struct edge{
	int v,nxt;
}a[5000005];
int head[10005],dfn[10005],low[10005],clr[10005],cnt,cnt2,tim;
stack<int> stk;
bool in[10005];
void add_edge(int u,int v){
	cnt++;
	a[cnt].v=v;
	a[cnt].nxt=head[u];
	head[u]=cnt;
}
void tarjan(int u){
	tim++;
	dfn[u]=low[u]=tim;
	stk.push(u);
	in[u]=1;
	for(int i=head[u];i;i=a[i].nxt){
		if(!dfn[a[i].v]){
			tarjan(a[i].v);
			low[u]=min(low[u],low[a[i].v]);
		}
		else if(in[a[i].v]){
			low[u]=min(low[u],dfn[a[i].v]);
		}
	}
	if(dfn[u]==low[u]){
		int k=stk.top();
		cnt2++;
		while(k!=u){
			clr[k]=cnt2;
			in[k]=0;
			stk.pop();
			k=stk.top();
		}
		clr[u]=cnt2;
		in[u]=0;
		stk.pop();
	}
}
int rd[10005],cd[10005];
int main(){
	int n,x;
	cin>>n;
	for(int i=1;i<=n;i++){
		cin>>x;
		while(x!=0){
			add_edge(i,x);
			cin>>x;
		}
	}
	for(int i=1;i<=n;i++)
		if(!dfn[i])
			tarjan(i);
	for(int i=1;i<=n;i++)
		for(int j=head[i];j;j=a[j].nxt)
			if(clr[i]!=clr[a[j].v]){
				rd[clr[a[j].v]]++;
				cd[clr[i]]++;
			}
	int sum1=0,sum2=0;
	for(int i=1;i<=cnt2;i++){
		if(rd[i]==0)sum1++;
		if(cd[i]==0)sum2++;
	}
	if(cnt2==1){
		cout<<1<<endl<<0;
		return 0;
	}
	cout<<sum1<<endl<<max(sum1,sum2);
	return 0;
}
