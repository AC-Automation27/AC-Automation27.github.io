#include<iostream>

int a[500005];

	int n,sum=0,kk;
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i],sum+=a[i];
	if(sum%3!=0){
		cout<<0;
		return 0;
	}
	kk=sum/3;
	sum=0;
	for(int i=1;i<=n;i++){
		sum+=a[i];
		if(sum==kk)b1[i]=1;
	}
	sum=0;

	}
	int ans=0;
	for(int i=3;i<=n;i++){
		ans+=b2[i]*c[i-2];
	}
	cout<<ans;
	return 0;
}

#include<iostream>
#include<cstdlib>
#include<cstdio>
using namespace std;
#define MAXN 1100005
struct fhq_treap{
	int rt,cnt,siz[MAXN],ch[MAXN][2],rnd[MAXN],val[MAXN];
	inline int new_node(int v){
		cnt++;
		rnd[cnt]=rand();
		val[cnt]=v;
		siz[cnt]=1;
		return cnt;
	}
	inline void up(int root){
		siz[root]=siz[ch[root][0]]+siz[ch[root][1]]+1;
	}
	int merge(int x,int y){
		if(!x||!y)return x+y;
		if(rnd[x]<rnd[y]){
			ch[x][1]=merge(ch[x][1],y);
			up(x);
			return x;
		}
		else{
			ch[y][0]=merge(x,ch[y][0]);
			up(y);
			return y;
		}
	}
	void split(int root,int pos,int &x,int &y){//<pos:x?y
		if(!root){
			x=y=0;
			return;
		}
		if(val[root]<pos){
			x=root;
			split(ch[root][1],pos,ch[root][1],y);
		}
		else{
			y=root;
			split(ch[root][0],pos,x,ch[root][0]);
		}
		up(root);
	}
	inline void insert(int val){
		int x,y=new_node(val),z;
		split(rt,val,x,z);
		rt=merge(merge(x,y),z);
	}
	inline void del(int val){
		int x,y,z;
		split(rt,val,x,y);
		split(y,val+1,y,z);
		y=merge(ch[y][0],ch[y][1]);
		rt=merge(merge(x,y),z);
	}
	inline int get_kth(int root,int k){
		while(1){
			if(siz[ch[root][0]]>=k)root=ch[root][0];
			else if(siz[ch[root][0]]==k-1)return val[root];
			else{
				k-=siz[ch[root][0]]+1;
				root=ch[root][1];
			}
		}
	}
	inline int get_rank(int v){
		int x,y,ans;
		split(rt,v,x,y);
		ans=siz[x]+1;
		rt=merge(x,y);
		return ans;
	}
	inline int pre(int v){
		int x,y,ans;
		split(rt,v,x,y);
		ans=get_kth(x,siz[x]);
		rt=merge(x,y);
		return ans;
	}
	inline int suf(int v){
		int x,y,ans;
		split(rt,v+1,x,y);
		ans=get_kth(y,1);
		rt=merge(x,y);
		return ans;
	}
}T; 
inline int read(){
	int x=0;
	char ch=getchar();
	while(ch<'0'||ch>'9')ch=getchar();
	while(ch>='0'&&ch<='9'){
		x*=10;
		x+=ch-48;
		ch=getchar();
	}
	return x;
}
int main(){
	int n,m,opt,x,lst=0,ans=0;
	cin>>n>>m;
	for(register int i=0;i<n;i++){
		opt=read();
		T.insert(opt);
	}
	for(register int i=0;i<m;i++){
		opt=read();
		x=read();
		x^=lst;
		if(opt==1){
			T.insert(x);
		}
		else if(opt==2){
			T.del(x);
		}
		else if(opt==3){
			lst=T.get_rank(x);
			ans^=lst;
		}
		else if(opt==4){
			lst=T.get_kth(T.rt,x);
			ans^=lst;
		}
		else if(opt==5){
			lst=T.pre(x);
			ans^=lst;
		}
		else if(opt==6){
			lst=T.suf(x);
			ans^=lst;
		}
	}
	cout<<ans<<endl;
	return 0;
}

#include<iostream>
#define int long long
using namespace std;
struct node{
	int v,nxt;
}a[400005];
int maxx[200005][2],sum[200005],pfs[200005];
int val[200005],head[200005],cnt;
void add_edge(int u,int v){
	cnt++;
	a[cnt].v=v;
	a[cnt].nxt=head[u];
	head[u]=cnt;
}
void dfs(int u,int fa){
	for(int i=head[u];i;i=a[i].nxt){
		if(maxx[u][0]==0)maxx[u][0]=val[a[i].v];
		else if(maxx[u][1]==0)maxx[u][1]=val[a[i].v];
		else if(val[a[i].v]>maxx[u][0]){maxx[u][1]=maxx[u][0];maxx[u][0]=val[a[i].v];}
		else if(val[a[i].v]>maxx[u][1])maxx[u][1]=val[a[i].v];
		pfs[u]+=val[a[i].v]*val[a[i].v];
		sum[u]+=val[a[i].v];
		if(a[i].v==fa)continue;
		dfs(a[i].v,u);
	}
} 
signed main(){
	int n,u,v,maxxx=0,ans=0;
	cin>>n;
	for(int i=1;i<n;i++){
		cin>>u>>v; 
		add_edge(u,v);
		add_edge(v,u);
	}
	for(int i=1;i<=n;i++)cin>>val[i];
	dfs(1,0);
	for(int i=1;i<=n;i++){
		sum[i]%=10007;
		pfs[i]%=10007;
		maxxx=max(maxxx,maxx[i][0]*maxx[i][1]);
		ans+=sum[i]*sum[i];
		ans-=pfs[i];
	}
	ans%=10007;
	ans+=10007;
	ans%=10007;
	cout<<maxxx<<" "<<ans;
	return 0;
}

#include<iostream>
#include<cmath>
#include<cstring>
using namespace std;
int c[100005]; 
int f(char ch1,char ch2){
	int x=ch1-96,y=ch2-96;
	return min(abs(x-y),min(abs(x+26-y),abs(x-y-26)));
} 
int main(){
	int n,k,ans=0;
	string s;
	cin>>n>>k>>s;
	k--;
	for(int i=0;i<n;i++){
		c[i]=f(s[i],s[n-i-1]);
	}
	int L=-1,R=-1;
	if(k<n/2){
		for(int i=0;i+i<=n;i++)if(c[i]){L=i;break;}
		for(int i=n/2-1;i>=0;i--)if(c[i]){R=i;break;}
		if(L==R&&L==-1){
			cout<<0;
			return 0;
		}
		if(k<=L){
			ans+=R-k;
		}
		else if(k>=R){
			ans+=k-L;
		}
		else{
			ans+=R-L+min(k-L,R-k); 
		}
		for(int i=L;i<=R;i++){
			ans+=c[i];
		}
		cout<<ans;
	}
	else{
		for(int i=n/2;i<=n;i++)if(c[i]){L=i;break;}
		for(int i=n;i>=n/2;i--)if(c[i]){R=i;break;}
		if(L==R&&L==-1){
			cout<<0;
			return 0;
		}
		if(k<=L){
			ans+=R-k;
		}
		else if(k>=R){
			ans+=k-L;
		}
		else{
			ans+=R-L+min(k-L,R-k); 
		}
		for(int i=L;i<=R;i++){

		cout<<ans;
		
	}
	return 0;
}
