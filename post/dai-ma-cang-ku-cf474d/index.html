#include<iostream>
#define int long long
const int mod=1000000007;
using namespace std;
int dp[500005],sum[500005];
int l[500005],r[500005];
signed main(){
	int n,k,maxx=0;
	cin>>n>>k;
	for(int i=0;i<n;i++){
		cin>>l[i]>>r[i];
		maxx=max(maxx,r[i]);
	}
	dp[0]=1;
	for(int i=1;i<=maxx;i++){
		if(i>=k){
			dp[i]=dp[i-1]+dp[i-k];
		}
		else dp[i]=dp[i-1];
		sum[i]=sum[i-1]+dp[i];
		sum[i]%=mod;
		dp[i]%=mod;
	}
	for(int i=0;i<n;i++){
		cout<<((sum[r[i]]-sum[l[i]-1])%mod+mod)%mod<<endl;
	}
	return 0;
}
