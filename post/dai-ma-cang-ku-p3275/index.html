#include<iostream>
#include<cstdio>
using namespace std;
int head[100005],p[100005],q[100000005],hd,tl;
long long dis[100005];
bool vis[100005];
int cnt;
struct node{
	int v,nxt,val;
}a[400005];
inline void add_edge(int u,int v,int val){
	a[++cnt].v=v;
	a[cnt].val=val;
	a[cnt].nxt=head[u];
	head[u]=cnt;
}
inline void read(int &x){
	char ch=getchar();
	x=0;
	while(ch<'0'||ch>'9')ch=getchar();
	while(ch>='0'&&ch<='9'){
		x=(x<<1)+(x<<3);
		x+=ch-48;
		ch=getchar(); 
	} 
}
signed main(){
	int n,m,u,v,w;
	cin>>n>>m;
	for(register int i=0;i<m;i++){
		read(w);read(v);read(u);
		if((w==2||w==4)&&u==v){
			cout<<-1;
			return 0;
		}
		if(w==1){
			add_edge(u,v,0);
			add_edge(v,u,0);
		}
		else if(w==2)
			add_edge(v,u,-1);
		else if(w==3)
			add_edge(u,v,0);
		else if(w==4)
			add_edge(u,v,-1);
		else if(w==5)
			add_edge(v,u,0);
	}
	for(register int i=1;i<=n;i++){
		q[tl++]=i;
	}
	while(hd<tl){
		u=q[hd];
		hd++;
		vis[u]=0;
		for(register int i=head[u];i;i=a[i].nxt){
			if(dis[u]+a[i].val<dis[a[i].v]){
				dis[a[i].v]=dis[u]+a[i].val;
				if(!vis[a[i].v])q[tl++]=a[i].v;
				vis[a[i].v]=1;
				p[a[i].v]++;
				if(p[a[i].v]==n){
					cout<<"-1";
					return 0;
				} 
			}
		}
	}
	long long sum=0;
	for(register int i=1;i<=n;i++)sum-=dis[i];
	cout<<sum+n;
	return 0;
}
