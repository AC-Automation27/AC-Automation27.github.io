#include<iostream>
#define int long long
using namespace std;
int a[500005];
bool b1[500005],b2[500005];
int c[500005];
signed main(){
	int n,sum=0,kk;
	cin>>n;
	for(int i=1;i<=n;i++)cin>>a[i],sum+=a[i];
	if(sum%3!=0){
		cout<<0;
		return 0;
	}
	kk=sum/3;
	sum=0;
	for(int i=1;i<=n;i++){
		sum+=a[i];
		if(sum==kk)b1[i]=1;
	}
	sum=0;
	for(int i=n;i>0;i--){
		sum+=a[i];
		if(sum==kk)b2[i]=1;
	}
	for(int i=1;i<=n;i++){
		c[i]=c[i-1]+b1[i];
	}
	int ans=0;
	for(int i=3;i<=n;i++){
		ans+=b2[i]*c[i-2];
	}
	cout<<ans;
	return 0;
}
