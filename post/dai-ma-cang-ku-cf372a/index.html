2020-02-28 【代码仓库】 

#include<iostream>
#include<algorithm>
using namespace std;
int a[500005];
bool book[500005]; 
int main(){
	int n;
	cin>>n;
	for(int i=0;i<n;i++)cin>>a[i];
	sort(a,a+n);
	int L=n/2-1,ans=0;
	for(int i=n-1;i>=0;i--){
		if(book[i])continue;
		while(book[L]||a[L]*2>a[i]){
			L--;
			if(L==-1)break;
		}
		if(L==-1)break;
		book[L]=1;
		ans++;
		book[i]=1;
	}
	cout<<n-ans;
	return 0;
}
