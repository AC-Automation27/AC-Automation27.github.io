#include<iostream>
#include<algorithm>
using namespace std;
struct lsh{
	int w,opt;
}data[200005];
bool operator<(lsh x,lsh y){
	return x.w<y.w;
}
int dd[200005],r[200005],root[200005],N,cnt=1;
void init(int n){
	for(int i=1;i<=n;i++)cin>>data[i].w,data[i].opt=i;
	sort(data+1,data+n+1);
	int temp=0;
	r[data[1].opt]=1;
	for(int i=2;i<=n;i++){
		if(data[i].w!=data[i-1].w){
			temp++;
			dd[temp]=data[i-1].w;
		}
		r[data[i].opt]=temp+1;
	}
	temp++;
	dd[temp]=data[n].w;
	N=temp;
}
struct node{
	int sum,ls,rs;
}a[20000005];
void build(int rt,int l,int r){
	if(l==r)return;
	int mid=(l+r)>>1;
	a[rt].ls=++cnt;
	build(cnt,l,mid);
	a[rt].rs=++cnt;
	build(cnt,mid+1,r);
}
void insert(int rt,int l,int r,int k,int real){
	int mid=(l+r)>>1;
	a[rt].ls=a[real].ls;a[rt].rs=a[real].rs;
	a[rt].sum=a[real].sum+1;
	if(l==r){
		return;
	}
	if(k<=mid){
		a[rt].ls=++cnt;
		insert(a[rt].ls,l,mid,k,a[real].ls);
	}
	else{
		a[rt].rs=++cnt;
		insert(a[rt].rs,mid+1,r,k,a[real].rs);
	}
}
int query(int rtl,int rtr,int l,int r,int k){
	int mid=(l+r)>>1;
	if(l==r)return l;
	int ss=a[a[rtr].ls].sum-a[a[rtl].ls].sum;
	if(ss>=k){
		return query(a[rtl].ls,a[rtr].ls,l,mid,k);
	}
	else{
		return query(a[rtl].rs,a[rtr].rs,mid+1,r,k-ss);
	}
}
int main(){
	int n,m,u,v,k;
	cin>>n>>m;
	init(n);
	root[0]=1;
	build(1,1,N);
	for(int i=1;i<=n;i++){
		root[i]=++cnt;
		insert(root[i],1,N,r[i],root[i-1]);
	}
	for(int i=1;i<=m;i++){
		cin>>u>>v>>k;
		cout<<dd[query(root[u-1],root[v],1,N,k)]<<endl;
	}
	return 0;
}
