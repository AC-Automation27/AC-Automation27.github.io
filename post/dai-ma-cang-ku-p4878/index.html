#include<iostream>
#define int long long
using namespace std;
int head[50005],dis[50005],queue[10000005],p[50005],hd,tl,cnt;
struct node{
	int v,nxt,val;
}a[1000005];
void add_edge(int u,int v,int val){
	a[++cnt].v=v;
	a[cnt].val=val;
	a[cnt].nxt=head[u];
	head[u]=cnt;
}
signed main(){
	int n,m1,m2,u,v,w;
	cin>>n>>m1>>m2;
	for(int i=0;i<m1;i++){
		cin>>u>>v>>w;
		add_edge(u,v,w);
	}
	for(int i=0;i<m2;i++){
		cin>>u>>v>>w;
		add_edge(v,u,-w);
	}
	for(int i=1;i<=n;i++){
		dis[i]=100000000000000000ll;
		add_edge(n+1,i,0);
	}
	queue[tl++]=n+1;
	while(hd<tl){
		u=queue[hd];
		hd++;
		for(int i=head[u];i;i=a[i].nxt){
			if(dis[u]+a[i].val<dis[a[i].v]){
				dis[a[i].v]=dis[u]+a[i].val;
				queue[tl++]=a[i].v;
				p[a[i].v]++;
				if(p[a[i].v]>n){
					cout<<-1;
					return 0;
				} 
			}
		}
	}
	for(int i=1;i<=n;i++)dis[i]=100000000000000000ll;
	dis[1]=0;
	hd=tl=0;
	queue[tl++]=1;
	while(hd<tl){
		u=queue[hd];
		hd++;
		for(int i=head[u];i;i=a[i].nxt){
			if(dis[u]+a[i].val<dis[a[i].v]){
				dis[a[i].v]=dis[u]+a[i].val;
				queue[tl++]=a[i].v;
			}
		}
	}
	if(dis[n]==100000000000000000ll){
		cout<<-2;
		return 0;
	}
	cout<<dis[n];
	return 0;
}
