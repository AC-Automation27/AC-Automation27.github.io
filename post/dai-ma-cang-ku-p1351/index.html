#include<iostream>
#define int long long
using namespace std;
struct node{
	int v,nxt;
}a[400005];
int maxx[200005][2],sum[200005],pfs[200005];
int val[200005],head[200005],cnt;
void add_edge(int u,int v){
	cnt++;
	a[cnt].v=v;
	a[cnt].nxt=head[u];
	head[u]=cnt;
}
void dfs(int u,int fa){
	for(int i=head[u];i;i=a[i].nxt){
		if(maxx[u][0]==0)maxx[u][0]=val[a[i].v];
		else if(maxx[u][1]==0)maxx[u][1]=val[a[i].v];
		else if(val[a[i].v]>maxx[u][0]){maxx[u][1]=maxx[u][0];maxx[u][0]=val[a[i].v];}
		else if(val[a[i].v]>maxx[u][1])maxx[u][1]=val[a[i].v];
		pfs[u]+=val[a[i].v]*val[a[i].v];
		sum[u]+=val[a[i].v];
		if(a[i].v==fa)continue;
		dfs(a[i].v,u);
	}
} 
signed main(){
	int n,u,v,maxxx=0,ans=0;
	cin>>n;
	for(int i=1;i<n;i++){
		cin>>u>>v; 
		add_edge(u,v);
		add_edge(v,u);
	}
	for(int i=1;i<=n;i++)cin>>val[i];
	dfs(1,0);
	for(int i=1;i<=n;i++){
		sum[i]%=10007;
		pfs[i]%=10007;
		maxxx=max(maxxx,maxx[i][0]*maxx[i][1]);
		ans+=sum[i]*sum[i];
		ans-=pfs[i];
	}
	ans%=10007;
	ans+=10007;
	ans%=10007;
	cout<<maxxx<<" "<<ans;
	return 0;
}
