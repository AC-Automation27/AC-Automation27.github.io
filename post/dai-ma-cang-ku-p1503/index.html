#include<iostream>
#include<cstdlib>
#include<stack>
using namespace std;
#define MAXN 100005
struct fhq_treap{
	int ch[MAXN][2],siz[MAXN],val[MAXN],rnd[MAXN],rt,cnt;
	void up(int root){
		siz[root]=siz[ch[root][0]]+siz[ch[root][1]]+1;
	}
	int new_node(int v){
		cnt++;
		val[cnt]=v;
		rnd[cnt]=rand();
		siz[cnt]=1;
		return cnt;
	}
	int merge(int x,int y){
		if(!x||!y){
			return x+y;
		}
		if(rnd[x]<rnd[y]){
			ch[x][1]=merge(ch[x][1],y);
			up(ch[x][1]);
			return x;
		}
		else{
			ch[y][0]=merge(x,ch[y][0]);
			up(ch[y][0]);
			return y;
		}
	}
	void split(int root,int pos,int &x,int &y){
		if(!root){
			x=y=0;
			return;
		}
		if(val[root]<pos){
			x=root;
			split(ch[x][1],pos,ch[x][1],y);
		}
		else{
			y=root;
			split(ch[y][0],pos,x,ch[y][0]);
		}
		up(root);
	}
	void insert(int val){
		int x,y;
		split(rt,val,x,y);
		rt=merge(merge(x,new_node(val)),y);
	}
	void del(int val){
		int x,y,z;
		split(rt,val,x,y);
		split(y,val+1,y,z);
		y=merge(ch[y][0],ch[y][1]);
		rt=merge(merge(x,y),z);
	}
	int rank(int v){
		int x,y,ret;
		split(rt,v,x,y);
		ret=siz[x];
		rt=merge(x,y);
		return ret+1;
	}
	int kth(int x,int k){
		while(1){
			if(siz[ch[x][0]]>=k){
				x=ch[x][0];
			}
			else if(siz[ch[x][0]]+1==k)return val[x];
			else{
				k-=siz[ch[x][0]]+1;
				x=ch[x][1];
			} 
		}
	}
	int pre(int val){
		int x,y,ret;
		split(rt,val,x,y);
		ret=kth(x,siz[x]);
		rt=merge(x,y);
		return ret;
	}
	int suf(int val){
		int x,y,ret;
		split(rt,val+1,x,y);
		ret=kth(y,1);
		rt=merge(x,y);
		return ret;
	}
} T;
int main(){
	int n,m,x;
	char ch;
	cin>>n>>m;
	T.insert(0);
	T.insert(n+1);
	stack<int> q;
	for(int i=0;i<m;i++){
		cin>>ch;
		if(ch=='D'){
			cin>>x;
			T.insert(x);
			q.push(x);
		}
		else if(ch=='R'){
			T.del(q.top());
			q.pop();
		}
		else{
			cin>>x;
			if(T.kth(T.rt,T.rank(x))==x)cout<<0<<endl;
			else cout<<T.suf(x)-T.pre(x)-1<<endl;
		}
	}
	return 0;
}
